if (operator === '+') {
return num1 + num2;
} else if (operator === '-') {
return num1 - num2;
} else if (operator === '/') {
return num1 / num2; } else if (operator === 'x') {

} else {
return 'Ingrese un parameto valido'
} }
console.log(calculator(2, '/', 2));
